CREATE TABLE RIDES (
    RIDE_ID NUMBER PRIMARY KEY,
    CAPTAIN_ID NUMBER REFERENCES CAPTAINS(CAPTAIN_ID),
    RIDE_DATE DATE NOT NULL,
    STATUS VARCHAR2(20) CHECK (STATUS IN ('Completed','Cancelled')),
    DISTANCE_km NUMBER(5,2) CHECK (DISTANCE_km > 0),
    EARNINGS NUMBER(8,2) CHECK (EARNINGS > 0),
    RATING NUMBER(2,1) CHECK (RATING BETWEEN 1 AND 5)
);
