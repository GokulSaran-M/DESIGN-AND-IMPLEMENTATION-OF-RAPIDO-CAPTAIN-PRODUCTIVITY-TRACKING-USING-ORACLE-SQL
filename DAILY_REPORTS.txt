CREATE TABLE DAILY_REPORTS (
    REPORT_DATE DATE,
    CAPTAIN_ID NUMBER REFERENCES CAPTAINS(CAPTAIN_ID),
    TOTAL_RIDES NUMBER,
    RIDES_COMPLETED NUMBER,
    HOURS_ACTIVE NUMBER(5,2),
    AVG_RATING NUMBER(3,2),
    TOTAL_EARNINGS NUMBER(8,2),
    PRIMARY KEY (REPORT_DATE, CAPTAIN_ID)
);
